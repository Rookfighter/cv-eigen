cmake_minimum_required(VERSION 3.5)
project(cv-eigen)

SET(CMAKE_CXX_STANDARD 11)
SET(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_MODULE_PATH ${CMAKE_MODULE_PATH} ${CMAKE_CURRENT_LIST_DIR}/cmake)

if(CMAKE_COMPILER_IS_GNUCXX)
    set(CMAKE_CXX_FLAGS "-Wall -Wextra")
endif(CMAKE_COMPILER_IS_GNUCXX)

# find OpenMP
find_package(OpenMP)
if(OPENMP_FOUND)
    set (CMAKE_C_FLAGS "${CMAKE_C_FLAGS} ${OpenMP_C_FLAGS}")
    set (CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} ${OpenMP_CXX_FLAGS}")
    set (CMAKE_EXE_LINKER_FLAGS "${CMAKE_EXE_LINKER_FLAGS} ${OpenMP_EXE_LINKER_FLAGS}")
endif(OPENMP_FOUND)

add_subdirectory(dep)

include_directories(
    include
    ${EIGEN3_INCLUDE_DIR}
    ${CATCH2_INCLUDE_DIR}
)

add_executable(detect_edges "examples/detect_edges.cpp")
add_executable(filter_smooth "examples/filter_smooth.cpp")
add_executable(optical_flow "examples/optical_flow.cpp")
add_executable(updown_sample "examples/updown_sample.cpp")

install(DIRECTORY "include/cve"
    DESTINATION include)


if(${BUILD_TESTS})
    # Compile tests
    add_executable(run_tests
        "test/main.cpp"
        # "test/test_kernel.cpp"
        # "test/test_sobel_operator.cpp"
        # "test/test_gaussian_filter.cpp"
    )
endif(${BUILD_TESTS})
